MosaicPixel

모자이크 픽셀의 크기를 설정합니다.

remarks

기본값은 "7" 입니다.

sample code

<script type="text/javascript" src="kphoto/js/raonkphoto.js"></script>	

<!-- ..... 생략 ..... -->
 
<div style="width:800px;height:200px">  
    <script type="text/javascript">
        var paramObj = {
            Id: 'photo1',

            // MosaicPixel을 15로 설정합니다.
            MosaicPixel: '15'
        };

        new RAONKPhoto(paramObj);
    </script>
</div>
